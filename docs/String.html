def html_strip
  self.gsub!(/\A---(.|\n)*?---/, '') # remove yaml front matter
  self.gsub!(/{%(.*)%}/, '') # remove functional liquid
  self.gsub!(%r{<\/?[^>]*>}, '') # remove html
  self.gsub!('\\n', '') # remove newlines
  self.gsub!(/\s+/, ' ') # remove extra space
  self.tr!('"', "'") # replace double quotes with single
  self
end
